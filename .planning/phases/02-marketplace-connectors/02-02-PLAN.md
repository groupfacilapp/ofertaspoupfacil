// src/lib/connectors/amazon.ts
import * as cheerio from 'cheerio';
import { createHash } from 'crypto';
import type {
  MarketplaceConnector,
  DecryptedCredentials,
  NormalizedOffer,
  FetchConfig,
  ValidationResult,
} from './types';
import { registerConnector } from './registry';

// --- Filters (exported for unit tests) ---

export const FILTER_DIGITAIS =
  /ebook|kindle|livro|capa comum|capa dura|audiolivro|audible|assinatura|prime video/i;

export const FILTER_ISBN = /^(85|65|978)\d{7}[\dX]$/i;

const FILTER_SPONSORED = /^Anúncio patrocinado\s?[–\-]\s?/i;

// --- Price utilities (exported for unit tests) ---

export function parsePrice(priceVal: unknown): number {
  if (priceVal === null || priceVal === undefined) return 0;
  if (typeof priceVal === 'number') return priceVal;
  let str = String(priceVal).trim();
  if (str.includes(',')) {
    str = str.replace(/\./g, '').replace(',', '.');
  }
  str = str.replace(/[^\d.]/g, '');
  return parseFloat(str) || 0;
}

// --- Offer filter (exported for unit tests) ---

export function shouldIncludeOffer(o: {
  currentPrice: number;
  originalPrice: number | null;
  title: string;
}): boolean {
  if (o.currentPrice === 0) return false;
  if (FILTER_DIGITAIS.test(o.title)) return false;
  if (FILTER_ISBN.test(o.title)) return false;
  // If there's no meaningful discount (currentPrice >= originalPrice - 15% tolerance), skip
  if (o.originalPrice !== null && o.currentPrice >= o.originalPrice * 1.15) return false;
  return true;
}

// --- Affiliate link (sync version exported for unit tests) ---

export function generateAmazonAffiliateLinkSync(productUrl: string, tag: string): string {
  const sep = productUrl.includes('?') ? '&' : '?';
  return `${productUrl}${sep}tag=${tag}`;
}

// --- Strategy 1: JSON Widget Parser (exported for unit tests) ---

export function parseAmazonStrategy1(html: string): NormalizedOffer[] {
  // Find any slot (e.g. slot-14) — do not hardcode slot number
  const slotMatch = html.match(/assets\.mountWidget\('slot-\d+',\s*/);
  if (!slotMatch || slotMatch.index === undefined) return [];

  const jsonStart = html.indexOf('{', slotMatch.index + slotMatch[0].length);
  if (jsonStart === -1) return [];

  // Extract balanced JSON object
  let depth = 0;
  let jsonEnd = -1;
  for (let i = jsonStart; i < html.length; i++) {
    if (html[i] === '{') depth++;
    else if (html[i] === '}') {
      depth--;
      if (depth === 0) {
        jsonEnd = i;
        break;
      }
    }
  }
  if (jsonEnd === -1) return [];

  let widgetData: any;
  try {
    widgetData = JSON.parse(html.slice(jsonStart, jsonEnd + 1));
  } catch {
    return [];
  }

  const versionedData = widgetData?.widgetData?.versionedWidgetData?.DEALS_V2;
  if (!versionedData) return [];

  const priceMap = versionedData.priceMap ?? {};
  const productMap = versionedData.productMap ?? {};
  const dealMap = versionedData.dealMap ?? {};
  const slotToDeals = versionedData.slotToDeals ?? {};

  // Collect all ASINs from all slots
  const allDealIds: string[] = Object.values(slotToDeals).flat() as string[];
  const offers: NormalizedOffer[] = [];

  for (const dealId of allDealIds) {
    const deal = dealMap[dealId];
    if (!deal) continue;
    const asin: string = deal.impressionAsin;
    if (!asin) continue;

    const product = productMap[asin];
    if (!product) continue;
    const priceInfo = priceMap[asin];

    // Title: remove sponsored prefix
    const rawTitle: string = product.title ?? '';
    const title = rawTitle.replace(FILTER_SPONSORED, '').trim();

    if (!title) continue;

    // Current price (check Pix/Boleto first)
    const activePromotions: any[] =
      product.promotionsUnified?.entity?.displayablePromotions ?? [];
    const pixOffer = activePromotions.find(
      (p: any) => p.combinedSavings?.fixedTargetAmount?.amount != null
    );

    let currentPriceRaw = 0;
    let condition: string | null = null;

    if (pixOffer) {
      currentPriceRaw = pixOffer.combinedSavings.fixedTargetAmount.amount;
      condition = 'À vista (Pix/Boleto)';
    } else {
      currentPriceRaw =
        priceInfo?.priceToPay?.moneyValueOrRange?.value?.amount ?? 0;
      if (deal.dealDetails?.entity?.type === 'LIGHTNING_DEAL') {
        condition = 'Oferta Relâmpago';
      }
    }

    const originalPriceRaw =
      priceInfo?.basisPrice?.moneyValueOrRange?.value?.amount ?? null;

    const currentPrice = Math.round(parsePrice(currentPriceRaw) * 100);
    const originalPrice =
      originalPriceRaw != null
        ? Math.round(parsePrice(originalPriceRaw) * 100)
        : null;

    // Installments from first buying option
    const buyingOptions: any[] = product.buyingOptions ?? [];
    let installments: string | null = null;
    if (buyingOptions.length > 0) {
      const inst = buyingOptions[0]?.installmentOptions?.displayableInstallment;
      if (inst?.numberOfInstallments && inst?.installmentAmount?.amount) {
        installments = `Parcelamento em ${inst.numberOfInstallments}x sem juros`;
      }
    }

    const imageUrl: string = product.primaryImage?.url ?? '';
    // Build full product URL
    const relativeUrl: string = product.detailPageURL ?? `/dp/${asin}`;
    const productUrl = relativeUrl.startsWith('http')
      ? relativeUrl
      : `https://www.amazon.com.br${relativeUrl}`;

    const offer: NormalizedOffer = {
      externalId: asin,
      marketplace: 'amazon',
      title,
      currentPrice,
      originalPrice,
      discountPercent:
        originalPrice && originalPrice > 0
          ? Math.round(((originalPrice - currentPrice) / originalPrice) * 100)
          : null,
      imageUrl,
      productUrl,
      affiliateLink: null, // generated separately
      condition,
      installments,
      category: null,
      sales: null,
    };

    if (shouldIncludeOffer({ currentPrice: offer.currentPrice, originalPrice: offer.originalPrice, title: offer.title })) {
      offers.push(offer);
    }
  }

  return offers;
}

// --- Strategy 2: HTML regex fallback ---

function parseAmazonStrategy2(html: string): NormalizedOffer[] {
  // Fallback: find data-component-type="s-search-result" divs
  const $ = cheerio.load(html);
  const offers: NormalizedOffer[] = [];

  $('[data-component-type="s-search-result"]').each((_i, el) => {
    const $el = $(el);
    const asin = $el.attr('data-asin');
    if (!asin) return;

    const title = $el.find('h2 a span').first().text().trim();
    if (!title || FILTER_DIGITAIS.test(title)) return;

    const priceWhole = $el.find('.a-price .a-price-whole').first().text().replace(/[^\d]/g, '');
    const priceFraction = $el.find('.a-price .a-price-fraction').first().text().replace(/[^\d]/g, '') || '00';
    const currentPriceCents = priceWhole ? parseInt(priceWhole, 10) * 100 + parseInt(priceFraction.padEnd(2, '0'), 10) : 0;
    if (currentPriceCents === 0) return;

    const imageUrl = $el.find('img.s-image').attr('src') ?? '';
    const relUrl = $el.find('h2 a').attr('href') ?? `/dp/${asin}`;
    const productUrl = relUrl.startsWith('http')
      ? relUrl
      : `https://www.amazon.com.br${relUrl}`;

    offers.push({
      externalId: asin,
      marketplace: 'amazon',
      title,
      currentPrice: currentPriceCents,
      originalPrice: null,
      discountPercent: null,
      imageUrl,
      productUrl,
      affiliateLink: null,
      condition: null,
      installments: null,
      category: null,
      sales: null,
    });
  });

  return offers;
}

**AliExpress API authentication note:**

The AliExpress Affiliate Open Platform API uses HMAC-SHA256 for request signing. The base URL is `https://api-sg.aliexpress.com/sync`. Each request includes:
- `app_key`: platform-level credential from AliExpress Open Platform (stored per-user as `api_key` in credentials)
- `tracking_id`: user's affiliate tracking ID
- `sign`: HMAC-SHA256 signature over sorted parameters

The sign algorithm: concatenate sorted param key-value pairs into a string, then HMAC-SHA256 with app_secret. Example:
```
sign = HMAC-SHA256(app_secret, "app_key{app_key}method{method}param1{val1}...")
```

Per research OPEN QUESTION about AliExpress auth — implement with HMAC-SHA256 (the official AliExpress Open Platform standard). Store `api_key` and `app_secret` as platform-level env vars (`ALIEXPRESS_APP_KEY`, `ALIEXPRESS_APP_SECRET`) and only `tracking_id` per user. This matches how the n8n built-in node works (platform OAuth, user tracking ID).

import { createHmac } from 'crypto';
import type {
  MarketplaceConnector,
  DecryptedCredentials,
  NormalizedOffer,
  FetchConfig,
  ValidationResult,
} from './types';
import { registerConnector } from './registry';

const ALIEXPRESS_API_URL = 'https://api-sg.aliexpress.com/sync';

// --- Signature utility (exported for unit tests) ---

export function buildAliExpressSignature(
  appSecret: string,
  params: Record<string, string>
): string {
  // Sort params alphabetically, concatenate key+value, HMAC-SHA256
  const sortedKeys = Object.keys(params).sort();
  const signString = sortedKeys.map((k) => `${k}${params[k]}`).join('');
  return createHmac('sha256', appSecret)
    .update(signString)
    .digest('hex')
    .toUpperCase();
}

// --- Normalizer (exported for unit tests) ---

export function normalizeAliExpressProduct(raw: any): NormalizedOffer {
  const currentPrice = Math.round(parseFloat(String(raw.sale_price ?? '0')) * 100);
  const originalPrice = raw.original_price
    ? Math.round(parseFloat(String(raw.original_price)) * 100)
    : null;

  const discountStr = String(raw.discount ?? '').replace('%', '').trim();
  const discountPercent = discountStr ? parseInt(discountStr, 10) || null : null;

  return {
    externalId: String(raw.product_id ?? raw.productId ?? ''),
    marketplace: 'aliexpress',
    title: String(raw.product_title ?? raw.productTitle ?? ''),
    currentPrice,
    originalPrice,
    discountPercent,
    imageUrl: String(raw.product_main_image_url ?? raw.imageUrl ?? ''),
    productUrl: String(raw.product_detail_url ?? raw.productUrl ?? ''),
    affiliateLink: raw.promotion_link ?? raw.promotionLink ?? null,
    condition: null,
    installments: null,
    category: raw.first_level_category_name ?? null,
    sales: raw.lastest_volume ?? raw.sales ?? null,
  };
}

// --- AliExpressConnector ---

export class AliExpressConnector implements MarketplaceConnector {
  readonly marketplace = 'aliexpress' as const;

  private buildRequest(
    method: string,
    params: Record<string, string>,
    appKey: string,
    appSecret: string
  ): Record<string, string> {
    const timestamp = new Date()
      .toISOString()
      .replace('T', ' ')
      .slice(0, 19);

    const allParams: Record<string, string> = {
      method,
      app_key: appKey,
      timestamp,
      format: 'json',
      v: '2.0',
      sign_method: 'hmac-sha256',
      ...params,
    };

    allParams.sign = buildAliExpressSignature(appSecret, allParams);
    return allParams;
  }

  private async callApi(params: Record<string, string>): Promise<any> {
    const body = new URLSearchParams(params);
    const response = await fetch(ALIEXPRESS_API_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body: body.toString(),
    });
    if (!response.ok) throw new Error(`AliExpress API error: ${response.status}`);
    return response.json();
  }

  async fetchOffers(config: FetchConfig): Promise<NormalizedOffer[]> {
    const appKey = process.env.ALIEXPRESS_APP_KEY;
    const appSecret = process.env.ALIEXPRESS_APP_SECRET;
    if (!appKey || !appSecret) {
      throw new Error('ALIEXPRESS_APP_KEY and ALIEXPRESS_APP_SECRET env vars required');
    }

    const { credentials, page, keywords, minSales } = config;
    const trackingId = credentials.tracking_id;
    if (!trackingId) throw new Error('AliExpress tracking_id credential required');

    const params: Record<string, string> = {
      tracking_id: trackingId,
      page_no: String(page),
      page_size: '50',
      target_currency: 'BRL',
      target_language: 'PT',
      sort: 'LAST_VOLUME_DESC',
    };

    if (keywords.length > 0) {
      params.keywords = keywords[0];
    }

    const requestParams = this.buildRequest(
      'aliexpress.affiliate.hotproduct.query',
      params,
      appKey,
      appSecret
    );

    const data = await this.callApi(requestParams);
    const products: any[] =
      data?.aliexpress_affiliate_hotproduct_query_response?.resp_result?.result
        ?.products?.product ?? [];

    const offers = products
      .map(normalizeAliExpressProduct)
      .filter((o) => (o.sales ?? 0) >= (minSales ?? 100) && o.currentPrice > 0);

    return offers;
  }

  async generateAffiliateLink(
    productUrl: string,
    credentials: DecryptedCredentials
  ): Promise<string> {
    const appKey = process.env.ALIEXPRESS_APP_KEY;
    const appSecret = process.env.ALIEXPRESS_APP_SECRET;
    if (!appKey || !appSecret) return productUrl;

    const trackingId = credentials.tracking_id;
    if (!trackingId) return productUrl;

    const params: Record<string, string> = {
      tracking_id: trackingId,
      promotion_link_type: '2',
      source_values: productUrl,
    };

    const requestParams = this.buildRequest(
      'aliexpress.affiliate.link.generate',
      params,
      appKey,
      appSecret
    );

    try {
      const data = await this.callApi(requestParams);
      const link =
        data?.aliexpress_affiliate_link_generate_response?.resp_result?.result
          ?.promotion_links?.promotion_link?.[0]?.promotion_link;
      return link ?? productUrl;
    } catch {
      return productUrl;
    }
  }

  async validateCredentials(credentials: DecryptedCredentials): Promise<ValidationResult> {
    if (!credentials.tracking_id) {
      return { valid: false, error: 'TrackingID é obrigatório' };
    }

    const appKey = process.env.ALIEXPRESS_APP_KEY;
    const appSecret = process.env.ALIEXPRESS_APP_SECRET;
    if (!appKey) {
      return { valid: false, error: 'App Key é obrigatória (variável de ambiente ALIEXPRESS_APP_KEY)' };
    }
    if (!appSecret) {
      return { valid: false, error: 'App Secret não configurado (variável de ambiente ALIEXPRESS_APP_SECRET)' };
    }

    // Validate by calling getHotProducts with page_size=1
    try {
      const params: Record<string, string> = {
        tracking_id: credentials.tracking_id,
        page_no: '1',
        page_size: '1',
        target_currency: 'BRL',
        target_language: 'PT',
        sort: 'LAST_VOLUME_DESC',
      };
      const requestParams = this.buildRequest(
        'aliexpress.affiliate.hotproduct.query',
        params,
        appKey,
        appSecret
      );
      const data = await this.callApi(requestParams);
      // A valid response has the response wrapper key
      const hasResponse =
        'aliexpress_affiliate_hotproduct_query_response' in data;
      return hasResponse
        ? { valid: true }
        : { valid: false, error: 'Resposta inválida da API AliExpress' };
    } catch (e) {
      return { valid: false, error: `Falha ao validar credenciais AliExpress: ${String(e)}` };
    }
  }
}

import { createClient as createAdminClient } from '@supabase/supabase-js';

    connectionId: string;
    config: Omit<FetchConfig, 'credentials'>;

      .from('marketplace_connections')
      .select('encrypted_credentials')
      .eq('id', connectionId)
      .eq('user_id', userId)

          affiliate_link: o.affiliateLink,

          installments: o.installments,
          category: o.category,
          fetched_at: new Date().toISOString(),
          expires_at: new Date(Date.now() + 48 * 60 * 60 * 1000).toISOString(),
        })),
        { onConflict: 'user_id,marketplace,external_id' }
      );
