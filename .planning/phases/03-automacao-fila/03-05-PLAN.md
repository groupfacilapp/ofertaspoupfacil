1. Create `src/app/api/cron/automation/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { supabaseAdmin } from '@/lib/supabase/admin';
import { tasks } from '@trigger.dev/sdk/v3';
import type { autoFetchTask } from '@/trigger/auto-fetch';
import type { autoDispatchTask } from '@/trigger/auto-dispatch';

const CRON_SECRET = process.env.CRON_SECRET;

export async function GET(req: NextRequest) {
  const authHeader = req.headers.get('authorization');
  if (CRON_SECRET && authHeader !== `Bearer ${CRON_SECRET}`) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  const brtHour = (new Date().getUTCHours() - 3 + 24) % 24;

  // 1. Find all active fetch rules that need to run
  const { data: fetchRules } = await supabaseAdmin
    .from('automation_rules')
    .select('id, user_id, marketplace, interval_minutes, last_run_at')
    .eq('rule_type', 'fetch')
    .eq('is_active', true);

  const fetchTriggered = [];
  for (const rule of fetchRules ?? []) {
    // Check interval
    if (rule.last_run_at) {
      const elapsed = (Date.now() - new Date(rule.last_run_at).getTime()) / 60000;
      if (elapsed < rule.interval_minutes) continue;
    }
    // Trigger the task
    await tasks.trigger<typeof autoFetchTask>('auto-fetch', {
      ruleId: rule.id,
      userId: rule.user_id,
      marketplace: rule.marketplace,
    });
    fetchTriggered.push(rule.id);
  }

  // 2. Find all active dispatch rules within time window
  const { data: dispatchRules } = await supabaseAdmin
    .from('automation_rules')
    .select('id, user_id, marketplace, interval_minutes, start_hour, end_hour, target_group_id, last_run_at')
    .eq('rule_type', 'dispatch')
    .eq('is_active', true);

  const dispatchTriggered = [];
  for (const rule of dispatchRules ?? []) {
    // Check time window
    if (brtHour < rule.start_hour || brtHour >= rule.end_hour) continue;
    // Check interval
    if (rule.last_run_at) {
      const elapsed = (Date.now() - new Date(rule.last_run_at).getTime()) / 60000;
      if (elapsed < rule.interval_minutes) continue;
    }
    // Must have a target group
    if (!rule.target_group_id) continue;
    // Trigger the task
    await tasks.trigger<typeof autoDispatchTask>('auto-dispatch', {
      ruleId: rule.id,
      userId: rule.user_id,
      marketplace: rule.marketplace,
      groupId: rule.target_group_id,
    });
    dispatchTriggered.push(rule.id);
  }

  return NextResponse.json({
    ok: true,
    hour: brtHour,
    fetchTriggered: fetchTriggered.length,
    dispatchTriggered: dispatchTriggered.length,
  });
}
```
